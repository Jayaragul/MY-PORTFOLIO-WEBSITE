import matplotlib.pyplot as plt

# Sample data
months = ['January', 'February', 'March', 'April', 'May', 'June']
values = [10, 20, 15, 30, 25, 40]

# Create a line chart
plt.figure(figsize=(10, 5))
plt.plot(months, values, marker='o', color='#50c8ff', linestyle='-', linewidth=2, markersize=8)
plt.fill_between(months, values, color='rgba(80, 200, 255, 0.2)')
plt.title('Monthly Data')
plt.xlabel('Month')
plt.ylabel('Values')
plt.grid(True)
plt.savefig('line_chart.png')
plt.show()
